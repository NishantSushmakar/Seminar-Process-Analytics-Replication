SELECT "order"."id" AS "case_id", 'order' AS "activity_id", CAST("order"."creation_date" AS VARCHAR) AS "timestamp"
FROM "order"
UNION ALL
SELECT "element_relation"."order" AS "case_id", 'invoice' AS "activity_id", CAST("invoice"."creation_date" AS VARCHAR) AS "timestamp"
FROM "element_relation"
JOIN "invoice" ON "element_relation"."invoice" = "invoice"."id"
ORDER BY "case_id", "timestamp";
