SELECT 
    CAST("CustomerID" AS VARCHAR) AS case_id, 
    CAST('Click_Logged_In' AS VARCHAR) AS activity_id, 
    CAST("ClickTimestamp" AS VARCHAR) AS timestamp 
FROM 
    "BPI2016_Clicks_Logged_In"
UNION ALL
SELECT 
    CAST("SessionID" AS VARCHAR) AS case_id, 
    CAST('Click_NOT_Logged_In' AS VARCHAR) AS activity_id, 
    CAST("ClickTimestamp" AS VARCHAR) AS timestamp 
FROM 
    "BPI2016_Clicks_NOT_Logged_In"
UNION ALL
SELECT 
    CAST("CustomerID" AS VARCHAR) AS case_id, 
    CAST('Complaint' AS VARCHAR) AS activity_id, 
    CAST("ComplaintDate" AS VARCHAR) AS timestamp 
FROM 
    "BPI2016_Complaints"
UNION ALL
SELECT 
    CAST("CustomerID" AS VARCHAR) AS case_id, 
    CAST('Question' AS VARCHAR) AS activity_id, 
    CAST("QuestionTimestampStart" AS VARCHAR) AS timestamp 
FROM 
    "BPI2016_Questions"
UNION ALL
SELECT 
    CAST("CustomerID" AS VARCHAR) AS case_id, 
    CAST('Werkmap_Message' AS VARCHAR) AS activity_id, 
    CAST("WerkmapTimestamp" AS VARCHAR) AS timestamp 
FROM 
    "BPI2016_Werkmap_Messages";
