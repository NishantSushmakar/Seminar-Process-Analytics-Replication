SELECT 
    "customer" AS case_id, 
    'order_creation_date' AS activity_id, 
    CAST("creation_date" AS TEXT) AS timestamp 
FROM 
    "order" 
WHERE 
    "customer" IS NOT NULL 
    AND "creation_date" IS NOT NULL

UNION ALL

SELECT 
    "customer" AS case_id, 
    'shipment_creation_date' AS activity_id, 
    CAST("creation_date" AS TEXT) AS timestamp 
FROM 
    "shipment" 
WHERE 
    "customer" IS NOT NULL 
    AND "creation_date" IS NOT NULL;
