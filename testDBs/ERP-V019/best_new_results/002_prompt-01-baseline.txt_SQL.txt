SELECT "customer"."id" AS "case_id", 'Order Created' AS "activity_id", CAST("order"."creation_date" AS VARCHAR) AS "timestamp"
FROM "order"
JOIN "customer" ON "order"."customer" = "customer"."id"
UNION ALL
SELECT "customer"."id" AS "case_id", 'Shipment Created' AS "activity_id", CAST("shipment"."creation_date" AS VARCHAR) AS "timestamp"
FROM "shipment"
JOIN "customer" ON "shipment"."customer" = "customer"."id";
