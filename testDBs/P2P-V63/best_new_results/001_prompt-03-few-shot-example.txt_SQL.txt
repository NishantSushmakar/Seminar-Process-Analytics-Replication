UNION ALL
SELECT "invoice"."order_id" AS "case_id", 'invoice' AS "activity_id", CAST("invoice"."creation_date" AS VARCHAR) AS "timestamp" FROM "invoice"
UNION ALL
SELECT "payment"."invoice_id" AS "case_id", 'payment' AS "activity_id", CAST("payment"."creation_date" AS VARCHAR) AS "timestamp" FROM "payment"
UNION ALL
SELECT "shipment"."order_id" AS "case_id", 'shipment' AS "activity_id", CAST("shipment"."creation_date" AS VARCHAR) AS "timestamp" FROM "shipment";
