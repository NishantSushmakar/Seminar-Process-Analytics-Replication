SELECT DISTINCT "order"."id" AS "case_id", 'order' AS "activity_id", CAST("order"."creation_date" AS VARCHAR) AS "timestamp"
FROM "order"
UNION
SELECT DISTINCT "shipment"."order_id" AS "case_id", 'shipment' AS "activity_id", CAST("shipment"."creation_date" AS VARCHAR) AS "timestamp"
FROM "shipment"
JOIN "order" ON "shipment"."order_id" = "order"."id"
UNION
SELECT DISTINCT "element_relation"."order" AS "case_id", 'invoice' AS "activity_id", CAST("invoice"."creation_date" AS VARCHAR) AS "timestamp"
FROM "invoice"
JOIN "element_relation" ON "invoice"."order_id" = "element_relation"."order"
UNION
SELECT DISTINCT "element_relation"."order" AS "case_id", 'payment' AS "activity_id", CAST("payment"."creation_date" AS VARCHAR) AS "timestamp"
FROM "payment"
JOIN "invoice" ON "payment"."invoice_id" = "invoice"."id"
JOIN "element_relation" ON "invoice"."order_id" = "element_relation"."order";
