SELECT DISTINCT CAST("CustomerID" AS VARCHAR) AS "case_id", CAST("PAGE_NAME" AS VARCHAR) AS "activity_id", CAST("ClickTimestamp" AS VARCHAR) AS "timestamp"
FROM "BPI2016_Clicks_Logged_In"
WHERE "CustomerID" IS NOT NULL AND "PAGE_NAME" IS NOT NULL AND "ClickTimestamp" IS NOT NULL
UNION
SELECT DISTINCT CAST("BPI2016_Clicks_Logged_In"."CustomerID" AS VARCHAR) AS "case_id", CAST("BPI2016_Clicks_NOT_Logged_In"."PAGE_NAME" AS VARCHAR) AS "activity_id", CAST("BPI2016_Clicks_NOT_Logged_In"."ClickTimestamp" AS VARCHAR) AS "timestamp"
FROM "BPI2016_Clicks_NOT_Logged_In"
JOIN "BPI2016_Clicks_Logged_In" ON "BPI2016_Clicks_NOT_Logged_In"."SessionID" = "BPI2016_Clicks_Logged_In"."SessionID"
WHERE "BPI2016_Clicks_Logged_In"."CustomerID" IS NOT NULL AND "BPI2016_Clicks_NOT_Logged_In"."PAGE_NAME" IS NOT NULL AND "BPI2016_Clicks_NOT_Logged_In"."ClickTimestamp" IS NOT NULL
